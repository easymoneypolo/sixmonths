	<script>
		const answers = document.querySelectorAll('.answer');

		answers.forEach(answer => {
			answer.addEventListener('click', () => {
				const isCorrect = answer.dataset.isCorrect === 'true';
				if (isCorrect) {
					answer.classList.add('correct');
					answer.parentNode.querySelector('.photo-container').style.display = 'block';
					const nextQuestionContainer = answer.parentNode.nextElementSibling;
					if (nextQuestionContainer) {
						nextQuestionContainer.style.display = 'block';
					}
				} else {
					answer.classList.add('incorrect');


			}
		});
	});
</script>
